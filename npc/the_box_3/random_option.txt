		while((.@optId == 86)
		|| (.@optId == 173)
		|| (.@optId == 174)
		|| (.@optId == 204)
		|| (.@optId == 205))
		.@optId = rand(1,254);
		
		freeloop(0);
		
		return .@optId;
	}
	
	while(((.@optId >= 76) && (.@optId <= 86))
	|| ((.@optId >= 173) && (.@optId <= 184))
	|| (.@optId == 204)
	|| (.@optId == 205))
	.@optId = rand(1,254);

	freeloop(0);
	
	return .@optId;
}
