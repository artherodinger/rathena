function	script	RefreshMystrian	{
	// Remove old mystrian buff
	bonus_script_clear 1;
	
	.@sumElement = "Ele_Neutral";
	if(mysteryElement$ != "" && mysteryElement$ != " ")
		.@sumElement = mysteryElement$;
	// Add new mystrian buff
	bonus_script "{ bonus2 bExpAddClass,Class_All,-100; bonus bAtkEle," + .@sumElement + "; }",86400;
	return;
}
