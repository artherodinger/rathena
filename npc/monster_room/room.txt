OnMonsterDead:

	.@allWeaponDropChance = 10;
	
	// All MvP map check
	if(strcharinfo(3) == $mr20name$){
		.@allWeaponDropChance += 40;
		mvpMapKiller++;
	}
	
	.@rand = rand(0,100);
	
	//debugmes "Chance: " + .@allWeaponDropChance;
	//debugmes "Random: " + .@rand;
	
	if(.@rand <= .@allWeaponDropChance){
		getmapxy(.@mn$,.@x,.@y);
		makeitem $allWeaponId[rand(0,$allWeaponIdArraySize - 1)],1,strcharinfo(3),.@x,.@y;
	}
	
	for(.@i = 1; .@i <= 20; .@i++){
		if(strcharinfo(3) == getd("$mr" + .@i + "name$")){
			set getd("$killed_mr" + .@i),getd("$killed_mr" + .@i) + 1;
			if(getd("$killed_mr" + .@i) >= getd("$mr" + .@i)){
				set getd("$killed_mr" + .@i),0;
				if(getd("$mr" + .@i) < 300)
				set getd("$mr" + .@i),getd("$mr" + .@i) + 30;

				callfunc("SpawnMonsterInRoom",.@i);
			}
			break;
		}
	}
	//debugmes "On map: " + strcharinfo(3) + ", Killed: " + getd("$killed_mr" + .@i) + "/" + getd("$mr" + .@i);
	end;
}
