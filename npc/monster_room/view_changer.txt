		if(Zeny < .priceHead1)

		changelook LOOK_HEAD_TOP,.@i;
	}
	@head1 = .@i;
	end;
	
	L_Head2:
	if(!head2Bought){
		menu "จ่ายเงิน",-;
		if(Zeny < .priceHead2)

		changelook LOOK_HEAD_MID,.@i;
	}
	@head2 = .@i;
	end;

		Zeny -= .priceHead3;

	if(.@i >= .lowestHat && .@i <= .highestHat)
