	end;
	
OnPCDieEvent:
	if(strcharinfo(3) != "prontera"){
		clearitem;
		.@maxDecreaseLevel = BaseLevel/10;
		if(.@maxDecreaseLevel <= 1)
		.@maxDecreaseLevel = 1;
		if(BaseLevel>1)
		BaseLevel -= rand(1,.@maxDecreaseLevel);
		statusup2 bStr,-1;
		statusup2 bAgi,-1;
		statusup2 bVit,-1;
		statusup2 bInt,-1;
		statusup2 bDex,-1;
		statusup2 bLuk,-1;
	}
	end;
}
