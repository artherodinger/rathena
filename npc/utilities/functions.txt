function	script	F_PercentBase10000	{
	.@v = getarg(0,0);
	if(.@v < 10)
	return "0.0" + .@v + "%";
	else if(.@v < 100)
	return "0." + .@v + "%";
	else{
		.@percent = (.@v / 100);
		.@decimal = (.@v % 100) / 10;
		return .@percent + "." + .@decimal + "%";
	}
	
	return;
}

function	script	F_IsDummyMonster	{
	.@v = getarg(0,0);
	if((.@v == 2408)			// Dummy
			|| (.@v == 2409)	// Dummy
			|| (.@v == 2410)	// Dummy
			|| (.@v == 2411)	// Dummy
			|| (.@v == 2413)	// Dummy
			|| (.@v == 2401))	// Poring Initial Land Map
	return true;
	else
	return false;
}
